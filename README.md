void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

// Initializing the LCD 
  lcd.init();            // Initialize the LCD
  lcd.backlight();       // turns on the backlight 
	
	lcd.setCursor(0,0); 
  lcd.print("Congrats to CWS");
  lcd.setCursor(0,1);
  lcd.print("Class of 2025!!");

  // Initialize OLED
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306 allocation failed"));
    for(;;);
  }
  display.clearDisplay();
  display.display();

}

void loop() {
  Serial.begin(9600);

// Trigger ultrasonic sensor
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2; // Convert to cm

  Serial.print("Distance: ");
  Serial.println(distance);

  display.clearDisplay();

  if (distance > 0 && distance <= 20) {
    // Show Tim bitmap
    display.drawBitmap(0, 0, timBitmap, 128, 64, SSD1306_WHITE);
	
	lcd.init();            // Initialize the LCD again as a loop 
  lcd.backlight();       // turns on the backlight 
	
	lcd.setCursor(0,0); 
  lcd.print("Loading Next...");
  lcd.setCursor(0,1);
  lcd.print("Destination: MIT!");
  } else {
    // Show Mermaid bitmap
    display.drawBitmap(0, 0, mermaidBitmap, 128, 64, SSD1306_WHITE);
	
	lcd.init();            // Initialize the LCD again as a loop 
  lcd.backlight();       // turns on the backlight 
	
	lcd.setCursor(0,0); 
  lcd.print("Congrats to CWS");
  lcd.setCursor(0,1);
  lcd.print("Class of 2025!!");
  
	

  }

  display.display();
  delay(200);
}
